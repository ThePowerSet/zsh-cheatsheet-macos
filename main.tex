\subsection*{Navigation}

\shortcut{\key{Ctrl} + \key{A}}{Jump to line start}
\shortcut{\key{Ctrl} + \key{E}}{Jump to line end}
\shortcut{\key{Ctrl} + \key{F}}{Move right}
\shortcut{\key{Ctrl} + \key{B}}{Move left}
\shortcut{\key{Esc}, \key{F}}{Next word}
\shortcut{\key{Esc}, \key{B}}{Previous word}
\shortcut{\key{Ctrl} + \key{T}}{Swap characters}
\shortcut{\key{Esc}, \key{T}}{Swap words}

\vspace{0.5em}

\subsection*{Editing}

\shortcut{\key{Ctrl} + \key{U}}{Delete to line start}
\shortcut{\key{Ctrl} + \key{K}}{Delete to line end}
\shortcut{\key{Ctrl} + \key{H}}{Delete previous character}
\shortcut{\key{Ctrl} + \key{W}}{Delete previous word}
\shortcut{\key{Esc}, \key{Backspace}}{Delete previous word}
\shortcut{\key{Ctrl} + \key{Y}}{Paste removed text}
\shortcut{\key{Ctrl} + \key{V}}{Insert literal character}
\shortcut{\key{Ctrl} + \key{\_}}{Undo last change}

\columnbreak

\subsection*{History}

\shortcut{\key{Ctrl} + \key{P}}{Previous command}
\shortcut{\key{Ctrl} + \key{N}}{Next command}
\shortcut{\key{Ctrl} + \key{R}}{Search backward}
\shortcut{\key{Ctrl} + \key{S}}{Search forward}
\shortcut{\key{Esc}, \key{P}}{Prefix match backward}
\shortcut{\key{Esc}, \key{N}}{Prefix match forward}
\shortcut{\key{Up} / \key{Down}}{Browse history}

\subsection*{Process control}

\shortcut{\key{Ctrl} + \key{Z}}{Suspend process}
\shortcut{\key{Ctrl} + \key{O}}{Run command, keep line}
\shortcut{\key{Ctrl} + \key{G}}{Send break}

\subsection*{Shell helpers}

\shortcut{\texttt{fg}}{Bring job to foreground}
\shortcut{\texttt{bg}}{Resume job in background}
\shortcut{\texttt{!!}}{Repeat last command}
\shortcut{\texttt{!\$}}{Reuse last argument}
